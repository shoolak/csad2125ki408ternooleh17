303int minimax(char currentPlayer, char aiPlayer, int depth) {
304 if (checkWin(aiPlayer)) {
305 return 10 - depth; // AI wins
306 } else if (checkWin(opponent(aiPlayer))) {
307 return depth - 10; // Opponent wins
308 } else if (isBoardFull()) {
309 return 0; // Draw
310 }
311 int bestScore = (currentPlayer == aiPlayer) ? -1000 : 1000;
312
313 for (int i = 0; i < BOARD_SIZE; i++) {
314 for (int j = 0; j < BOARD_SIZE; j++) {
315 if (board[i][j] != PLAYER_X && board[i][j] != PLAYER_O) {
316 board[i][j] = currentPlayer; // Make the move
317 int score = minimax(opponent(currentPlayer), aiPlayer, depth + 1);
318 board[i][j] = '1' + (i * BOARD_SIZE + j); // Reset the move
319 if (currentPlayer == aiPlayer) {
320 if (score > bestScore) {
321 bestScore = score;
322 }
323 } else {
324 if (score < bestScore)
325 {
326 bestScore = score;
327 }
328 }
329 }

331 }
